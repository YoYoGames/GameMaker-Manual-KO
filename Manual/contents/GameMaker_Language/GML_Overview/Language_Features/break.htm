break

그만큼 break 문은 조기 종료하는 데 사용됩니다.for,repeat,while 또는do / until 어떤 종류의 루프, 또는switch 그 시점에서 끝내거나 조기에 끝내는 진술with 전화. 이러한 다양한 기능에 대한 개별 페이지를 참조하여 각 상황에서 이 기능을 사용하는 방법에 대한 자세한 설명을 확인하십시오. 경우에 유의하십시오. break 이러한 컨텍스트 외부에서 사용되면 오류가 발생합니다.

아래에서 방법에 대한 몇 가지 기본 예를 볼 수 있습니다. break 사용할 수 있으며 구문은 간단합니다.

break;

break 안에 for 고리:

for (var i = 0; i < 10; i += 1)
{
    if (array[i] == 234)
    {
        break;
    }
}

num = i;

break 안에 repeat 고리:

var i = 0;
var temp = 0;
repeat (10)
{
    temp += array[i];
    if (temp > max_total)
    {
        break;
    }
    else
    {
        i += 1;
    }
}

break 안에 while 고리:

var i = 0;
while (!place_free(x, y))
{
    x = random(room_width);
    y = random(room_height);
    if (i > 50)
    {
        break;
    }
    else
    {
        i+=1;
    }
}

break 안에 do / until 고리:

var _id = noone;
do
{
    _id = list[| 0];
    if (instance_exists(_id))
    {
        break;
    }
    ds_list_delete(list, 0);
}
until (ds_list_empty(list));
target = _id;

break 사용할 때 with:

var count = 0;
with (obj_Enemy)
{
    count++;
    if (count > 10)
    {
        break;
    }
    hp = 100;
}

break 안에 switch:

switch (keyboard_key)
{
    case vk_left:
    case ord("A"):
        x -= 4;
    break;

    case vk_right:
    case ord("D"):
        x += 4;
    break;

    case vk_up:
    case ord("W"):
        y -= 4;
    break;

    case vk_down:
    case ord("S"):
        y += 4;
    break;
}
