var _handle = call_later(5, time_source_units_seconds, _callback);

이 예에서는 5초가 지난 후 디버그 메시지를 표시하는 콜백 함수를 만듭니다. 그 후, 기본 시간 소스 가 파괴됩니다.

예 2:

var _create_enemy = function()
{
    var _x = random(room_width);
    var _y = random(room_height);
    instance_create_depth(_x, _y, depth, obj_enemy);
}

var _handle = call_later(10, time_source_units_seconds, _create_enemy, true);

이 예에서는 방의 임의 위치에 새로운 적을 생성하는 콜백 함수를 만듭니다. 그런 다음 이 함수는 call_later 에 전달하여 10초마다 실행됩니다. 마지막 인수는 true 로 설정되어 함수가 10초마다 계속 호출됩니다.

call_later 에서 반환된 핸들과 함께 call_cancel 을 사용하여 시간 소스 와 콜백 호출을 중지합니다.
